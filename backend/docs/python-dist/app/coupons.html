 189def fetch(url, timeout=None, retries=MAX_RETRIES, method='GET',
 190          data=None, json_data=None, headers_extra=None, session=None):
 191    """
 192    Executes an HTTP request with built-in retries, custom headers, and backoff delays.
 193
 194    This function attempts to fetch a given URL using either a provided session or 
 195    the standard requests library. It automatically injects browser-like headers 
 196    using `get_headers()`. If a request fails, it implements an escalating delay 
 197    (linear backoff) before retrying, up to the maximum number of retries.
 198
 199    Args:
 200        url (str): The target URL to fetch.
 201        timeout (int, optional): Request timeout in seconds. Defaults to the global TIMEOUT.
 202        retries (int, optional): Number of retry attempts upon failure. Defaults to MAX_RETRIES.
 203        method (str, optional): The HTTP method to use ('GET' or 'POST'). Defaults to 'GET'.
 204        data (dict, optional): Form data to send in the body of a POST request.
 205        json_data (dict, optional): JSON data to send in the body of a POST request.
 206        headers_extra (dict, optional): Additional HTTP headers to merge with the default ones.
 207        session (requests.Session, optional): An existing requests session to use for connection pooling/cookies.
 208
 209    Returns:
 210        requests.Response | None: The response object if successful, or None if all 
 211            retries are exhausted or a fatal exception occurs.
 212    """
 213    t = timeout or TIMEOUT
 214    requester = session if session else requests
 215    for attempt in range(retries + 1):
 216        try:
 217            h = get_headers()
 218            if headers_extra:
 219                h.update(headers_extra)
 220            if method == 'POST':
 221                resp = requester.post(url, headers=h, timeout=t,
 222                                      data=data, json=json_data, allow_redirects=True)
 223            else:
 224                resp = requester.get(url, headers=h, timeout=t, allow_redirects=True)
 225            return resp
 226        except requests.exceptions.RequestException as e:
 227            if attempt < retries:
 228                time.sleep(DELAY * (attempt + 1))
 229            else:
 230                print(f"  [!] Failed {url[:70]} -> {e}")
 231                return None

 233def extract_domain(url):
 234    """
 235    Extracts and cleans the root domain from a given URL.
 236
 237    Parses the provided URL to isolate the network location (netloc), converts 
 238    it to lowercase for consistency, and removes any leading 'www.' prefix. This 
 239    ensures that URLs like 'https://www.example.com/page' and 'http://example.com' 
 240    both resolve to the same standardized base domain ('example.com').
 241
 242    Args:
 243        url (str): The full URL string to process.
 244
 245    Returns:
 246        str: The cleaned, lowercase domain name without the 'www.' prefix.
 247    """
 248    parsed = urllib.parse.urlparse(url)
 249    domain = parsed.netloc.lower()
 250    return re.sub(r'^www\.', '', domain)

 252def extract_base(url):
 253    """
 254    Extracts the base URL (scheme and domain) from a full URL string.
 255
 256    Parses the provided URL to isolate the scheme (e.g., 'http' or 'https') 
 257    and the network location (e.g., 'www.example.com'). This is particularly 
 258    useful for reconstructing absolute URLs when scraping pages that use 
 259    relative links for internal resources or API endpoints.
 260
 261    Args:
 262        url (str): The full URL to parse.
 263
 264    Returns:
 265        str: The base URL in the format "scheme://netloc" (e.g., 
 266            "https://www.example.com").
 267    """
 268    p = urllib.parse.urlparse(url)
 269    return f"{p.scheme}://{p.netloc}"

 271def extract_brand_name(domain):
 272    """
 273    Attempts to extract the core brand name from a domain string.
 274
 275    This function uses a heuristic approach by splitting the domain into its 
 276    constituent parts (separated by dots). It iterates through these parts and 
 277    skips common generic subdomains and Top-Level Domains (TLDs) like 'www', 
 278    'shop', 'co', and 'uk'. The first remaining part that is longer than two 
 279    characters is returned as the brand name.
 280
 281    Args:
 282        domain (str): The domain string to parse (e.g., 'shop.nike.co.uk').
 283
 284    Returns:
 285        str: The extracted brand name (e.g., 'nike'). If no suitable part is 
 286            found, it falls back to returning the first part of the domain, 
 287            or the raw domain string itself.
 288    """
 289    parts = domain.split('.')
 290    skip = {'shop', 'store', 'www', 'us', 'uk', 'en', 'buy', 'get', 'co'}
 291    for p in parts:
 292        if p not in skip and len(p) > 2:
 293            return p
 294    return parts[0] if parts else domain

 296def extract_product_info(url, page_text=''):
 297    """
 298    Extracts product metadata (name, brand, category, and model) from an e-commerce page.
 299
 300    This function utilizes a robust cascade of fallback methods to parse the page, 
 301    ensuring maximum compatibility across different store platforms:
 302    1. JSON-LD structured data (most reliable, standard for modern e-commerce).
 303    2. Open Graph (`og:title`) and Twitter card meta tags.
 304    3. Standard HTML `<title>` tags.
 305    4. Common CSS selectors for product headings (e.g., 'h1', '.product-title').
 306    5. Brand inference (assumes the first word of the product name is the brand).
 307    6. Model number extraction via regex against the URL and product name.
 308
 309    Args:
 310        url (str): The full URL of the product page.
 311        page_text (str, optional): Pre-fetched raw HTML of the page. If omitted, 
 312            the function will attempt to fetch the HTML automatically.
 313
 314    Returns:
 315        dict: A dictionary containing the extracted data with keys:
 316            'product_name', 'product_brand', 'category', and 'model'. Missing 
 317            values will be represented as empty strings.
 318    """
 319    info = {'product_name': '', 'product_brand': '', 'category': '', 'model': ''}
 320
 321    if not page_text:
 322        # Try to fetch the page if we don't have it
 323        resp = fetch(url, timeout=10)
 324        if resp:
 325            page_text = resp.text
 326        else:
 327            return info
 328
 329    soup = BeautifulSoup(page_text, 'html.parser')
 330
 331    # 1. JSON-LD structured data (most reliable — used by most modern stores)
 332    for script in soup.find_all('script', type='application/ld+json'):
 333        try:
 334            data = json.loads(script.string or '{}')
 335            if isinstance(data, list):
 336                data = next((d for d in data if d.get('@type') == 'Product'), data[0] if data else {})
 337            if data.get('@type', '').lower() == 'product':
 338                info['product_name'] = data.get('name', '')
 339                brand = data.get('brand', {})
 340                info['product_brand'] = brand.get('name', '') if isinstance(brand, dict) else str(brand)
 341                info['category'] = data.get('category', '')
 342                info['model'] = data.get('model', data.get('mpn', data.get('sku', '')))
 343                if info['product_name']:
 344                    break
 345        except Exception:
 346            pass
 347
 348    # 2. Open Graph tags
 349    if not info['product_name']:
 350        og = soup.find('meta', property='og:title') or soup.find('meta', attrs={'name': 'twitter:title'})
 351        if og:
 352            info['product_name'] = og.get('content', '').split('|')[0].split(' - ')[0].strip()
 353
 354    # 3. Page <title>
 355    if not info['product_name']:
 356        title = soup.find('title')
 357        if title:
 358            info['product_name'] = title.get_text().split('|')[0].split(' - ')[0].strip()
 359
 360    # 4. Common product heading tags
 361    if not info['product_name']:
 362        for sel in ['h1', '[class*="product-title"]', '[class*="product-name"]',
 363                    '[class*="pdp-title"]', '[id*="product-title"]']:
 364            el = soup.select_one(sel)
 365            if el:
 366                info['product_name'] = el.get_text(strip=True)
 367                break
 368
 369    # 5. Extract brand from product name if not found yet
 370    if not info['product_brand'] and info['product_name']:
 371        # First word of product name is often the brand (Apple iPhone, LG Fridge, Samsung TV)
 372        first_word = info['product_name'].split()[0] if info['product_name'].split() else ''
 373        if len(first_word) >= 2 and first_word.isalpha():
 374            info['product_brand'] = first_word
 375
 376    # 6. Extract model number from URL + product name
 377    combined = url + ' ' + info['product_name']
 378    model_match = re.search(r'\b([A-Z]{1,5}[0-9]{2,}[A-Z0-9]{0,10})\b', combined.upper())
 379    if model_match and not info['model']:
 380        info['model'] = model_match.group(1)
 381
 382    return info

 411def filter_codes_by_product(codes, product_info):
 412    """
 413    Evaluates and categorizes discount codes based on their relevance to a specific product.
 414
 415    This function calculates a relevance score for each code using a heuristic point system:
 416    - Strong Positive (+5): Code contains the brand name or its abbreviation (e.g., 'SAM' for Samsung).
 417    - Positive (+3): Code contains a keyword matching the product's detected category.
 418    - Negative (-3): Code contains a keyword belonging to a clearly different category.
 419    - Neutral Boost (+1): Code contains generic discount terminology with numbers (e.g., 'SAVE20', '10OFF').
 420
 421    Based on the final score, codes are segregated into three distinct buckets.
 422
 423    Args:
 424        codes (list of str): A list of raw discount codes to be evaluated.
 425        product_info (dict): A dictionary containing metadata about the target product. 
 426            Expected keys include 'product_name', 'product_brand', and 'category'.
 427
 428    Returns:
 429        tuple: A 3-tuple containing lists of strings:
 430            - relevant (list of str): Codes with a score > 0, sorted descending by score.
 431            - neutral (list of str): Codes with a score of exactly 0.
 432            - irrelevant (list of str): Codes with a score < 0.
 433    """
 434    product_name = product_info.get('product_name', '').upper()
 435    product_brand = product_info.get('product_brand', '').upper()
 436    category = product_info.get('category', '').upper()
 437    combined_text = product_name + ' ' + product_brand + ' ' + category
 438
 439    # Detect which categories this product belongs to
 440    detected_categories = set()
 441    for cat, keywords in CATEGORY_KEYWORDS.items():
 442        if any(kw in combined_text for kw in keywords):
 443            detected_categories.add(cat)
 444
 445    # All keywords for detected categories
 446    relevant_keywords = []
 447    for cat in detected_categories:
 448        relevant_keywords.extend(CATEGORY_KEYWORDS[cat])
 449
 450    # All keywords for OTHER categories (negative signal)
 451    irrelevant_keywords = []
 452    for cat, keywords in CATEGORY_KEYWORDS.items():
 453        if cat not in detected_categories:
 454            irrelevant_keywords.extend(keywords)
 455
 456    # Brand abbreviations (e.g. LG -> LG, Samsung -> SAM, Apple -> APP)
 457    brand_abbrevs = set()
 458    if product_brand:
 459        brand_abbrevs.add(product_brand[:2])
 460        brand_abbrevs.add(product_brand[:3])
 461        brand_abbrevs.add(product_brand[:4])
 462        brand_abbrevs.add(product_brand)
 463
 464    relevant = []
 465    irrelevant = []
 466    neutral = []
 467
 468    for code in codes:
 469        c = code.upper()
 470        score = 0
 471
 472        # Strong positive: code contains product brand name/abbreviation
 473        for abbrev in brand_abbrevs:
 474            if len(abbrev) >= 2 and abbrev in c:
 475                score += 5
 476                break
 477
 478        # Positive: code contains a keyword from detected category
 479        for kw in relevant_keywords:
 480            if len(kw) >= 4 and kw[:4] in c:
 481                score += 3
 482                break
 483
 484        # Negative: code clearly belongs to a different category
 485        for kw in irrelevant_keywords:
 486            if len(kw) >= 4 and kw[:4] in c:
 487                score -= 3
 488                break
 489
 490        # Neutral boost: generic discount codes (SAVE20, 10OFF, BOOST25 etc.)
 491        if re.search(r'(SAVE|OFF|DEAL|EXTRA|CODE|BOOST|COOL|WINTER|SUMMER|FLASH)\d', c):
 492            score += 1
 493        if re.search(r'\d+(OFF|PCT|PERCENT)', c):
 494            score += 1
 495
 496        if score > 0:
 497            relevant.append((code, score))
 498        elif score < 0:
 499            irrelevant.append(code)
 500        else:
 501            neutral.append(code)
 502
 503    relevant.sort(key=lambda x: x[1], reverse=True)
 504    return [c for c, _ in relevant], neutral, irrelevant

 507def clean_codes(codes):
 508    """
 509    Sanitizes, filters, and deduplicates a raw list of scraped discount codes.
 510
 511    This function processes a list of strings through a strict set of rules to 
 512    weed out false positives (like common English words, pure numbers, or analytics tags). 
 513    It ensures uniqueness while preserving the original discovery order by using an `OrderedDict`.
 514
 515    Filtering criteria for a valid code:
 516    - Must be between MIN_CODE_LENGTH and 20 characters long.
 517    - Cannot be a Google Tag Manager ID (e.g., 'GTM-XXXX').
 518    - Cannot be in the predefined `JUNK_CODES` blacklist.
 519    - Cannot consist entirely of numbers.
 520    - Must contain at least one alphabetical letter.
 521    - Must either contain at least one digit OR be at least 6 characters long 
 522      (to filter out short, capitalized English words that slip past the blacklist).
 523
 524    Args:
 525        codes (list of str): The raw list of strings extracted from web elements or regex.
 526
 527    Returns:
 528        list of str: A cleaned, deduplicated list of valid, uppercase discount codes.
 529    """
 530    seen = OrderedDict()
 531    for c in codes:
 532        c = c.strip().upper()
 533        # Remove GTM/analytics tag patterns like GTM-XXXX
 534        if re.match(r'^GTM-', c):
 535            continue
 536        if (MIN_CODE_LENGTH <= len(c) <= 20
 537                and c not in JUNK_CODES
 538                and not re.match(r'^\d+$', c)
 539                and re.search(r'[A-Z]', c)
 540                # Must contain at least one digit OR be all caps 6+ chars
 541                # This filters out plain English words that slip through
 542                and (re.search(r'\d', c) or len(c) >= 6)):
 543            seen[c] = True
 544    return list(seen.keys())

 546def find_codes_in_text(text):
 547    """
 548    Extracts potential discount codes from a block of plain text using regular expressions.
 549
 550    This function iterates through a globally defined list of strict regex patterns 
 551    (`CODE_PATTERNS`). These patterns are designed to look for context clues—such as 
 552    "use code", "coupon:", or "save 20% with"—to accurately capture the actual discount 
 553    code nearby while ignoring random strings. The search is case-insensitive.
 554
 555    Args:
 556        text (str): The string content to parse. This could be raw HTML text, a JSON 
 557            snippet, or a paragraph from a webpage.
 558
 559    Returns:
 560        list of str: A raw list of matched code strings. Note that this list may 
 561            contain duplicates or junk codes until it is passed through `clean_codes()`.
 562    """
 563    codes = []
 564    for pattern in CODE_PATTERNS:
 565        matches = re.findall(pattern, text, re.IGNORECASE)
 566        codes.extend(matches)
 567    return codes

 569def score_code(code):
 570    """
 571    Calculates a baseline heuristic score for a standalone discount code.
 572
 573    This function evaluates a code based on common promotional patterns to 
 574    determine its likelihood of being a genuine, usable discount code. It 
 575    awards points based on the following criteria:
 576    - +3 points for each promotional keyword found (e.g., 'SAVE', 'FREE', 'VIP').
 577    - +2 points if the code contains at least one numeric digit.
 578    - +1 point if the code's length is within the typical "sweet spot" for 
 579      coupons (between 4 and 15 characters).
 580
 581    Args:
 582        code (str): The discount code string to evaluate.
 583
 584    Returns:
 585        int: The total calculated heuristic score for the code. Higher scores 
 586            indicate a higher probability of the string being a real promo code.
 587    """
 588    score = 0
 589    keywords = ['SAVE', 'OFF', 'FREE', 'DEAL', 'PROMO', 'CODE', 'COUPON',
 590                'DISCOUNT', 'FIRST', 'WELCOME', 'NEW', 'VIP', 'EXTRA', 'FLASH']
 591    for kw in keywords:
 592        if kw in code.upper():
 593            score += 3
 594    if re.search(r'\d', code):
 595        score += 2
 596    if 4 <= len(code) <= 15:
 597        score += 1
 598    return score

 600def extract_codes_from_html(html):
 601    """
 602    Extracts potential discount codes from rendered HTML using targeted CSS selectors and regex.
 603
 604    This function employs a two-pronged approach to find codes within a webpage:
 605    1. DOM Element Targeting: It uses BeautifulSoup to hunt for specific HTML attributes 
 606       (e.g., 'data-clipboard-text', 'data-coupon') and class/tag combinations (e.g., 
 607       'input[readonly]', '.promo-code') commonly used by e-commerce platforms to store 
 608       and display discount codes.
 609    2. Broad Regex Sweep: It parses the combined text and attributes of those targeted 
 610       elements, as well as the entire raw HTML document, using `find_codes_in_text()` 
 611       to catch any inline codes that don't sit inside neatly predictable containers.
 612
 613    All extracted candidates are finally passed through `clean_codes()` for strict 
 614    validation and deduplication.
 615
 616    Args:
 617        html (str): The raw or rendered HTML string to parse. Null bytes and 
 618            problematic characters are automatically stripped before parsing.
 619
 620    Returns:
 621        list of str: A cleaned, deduplicated list of valid discount codes extracted 
 622            from the HTML structure.
 623    """
 624    if not html or not isinstance(html, str):
 625        return []
 626    # Strip null bytes and other problematic chars
 627    html = html.replace('\x00', '').encode('utf-8', errors='replace').decode('utf-8', errors='replace')
 628    try:
 629        soup = BeautifulSoup(html, 'html.parser')
 630    except Exception:
 631        return find_codes_in_text(html)
 632    codes = []
 633
 634    selectors = [
 635        '[data-clipboard-text]', '[data-code]', '[data-coupon]', '[data-promo]',
 636        '[class*="coupon-code"]', '[class*="promo-code"]', '[class*="voucher-code"]',
 637        '[class*="code-text"]', '[class*="discount-code"]', '[class*="offer-code"]',
 638        'input[readonly]', 'span[class*="code"]', 'div[class*="code"]',
 639        'p[class*="code"]', 'strong[class*="code"]', '[class*="copy-code"]',
 640        '[class*="reveal"]', 'button[class*="code"]',
 641    ]
 642
 643    for sel in selectors:
 644        try:
 645            for el in soup.select(sel):
 646                combined = el.get_text(strip=True)
 647                for attr, val in el.attrs.items():
 648                    if isinstance(val, str):
 649                        combined += ' ' + val
 650                # Short standalone codes (like "AB10") directly in the element
 651                stripped = combined.strip().upper()
 652                if (3 <= len(stripped) <= 20
 653                        and re.match(r'^[A-Z0-9\-_]+$', stripped)
 654                        and stripped not in JUNK_CODES
 655                        and re.search(r'[A-Z]', stripped)):
 656                    codes.append(stripped)
 657                codes += find_codes_in_text(combined)
 658        except Exception:
 659            pass
 660
 661    codes += find_codes_in_text(html)
 662    return clean_codes(codes)

 665def detect_platform(base_url, page_text=''):
 666    """
 667    Attempts to identify the underlying e-commerce platform of a store.
 668
 669    This function uses a two-step fingerprinting approach. First, it actively probes 
 670    for Shopify by requesting the standard `/cart.js` endpoint. If that fails or 
 671    returns unexpected data, it falls back to passively scanning the provided HTML 
 672    source text for common platform-specific signatures (like 'wc-ajax' for 
 673    WooCommerce or 'mage/' for Magento).
 674
 675    Args:
 676        base_url (str): The root URL of the store (e.g., 'https://www.example.com').
 677        page_text (str, optional): The raw HTML string of the store's page to scan 
 678            for keywords. Defaults to an empty string.
 679
 680    Returns:
 681        str: The name of the detected platform ('shopify', 'woocommerce', 'magento', 
 682            'bigcommerce'). Returns 'unknown' if no recognizable signatures are found.
 683    """
 684    resp = fetch(f"{base_url}/cart.js", timeout=6)
 685    if resp and resp.status_code == 200:
 686        try:
 687            data = resp.json()
 688            if 'items' in data:
 689                return 'shopify'
 690        except Exception:
 691            pass
 692
 693    if not isinstance(page_text, str):
 694        page_text = str(page_text)
 695    text = page_text.lower()
 696    if 'wc-ajax' in text or 'woocommerce' in text:
 697        return 'woocommerce'
 698    if 'magento' in text or 'mage/' in text:
 699        return 'magento'
 700    if 'bigcommerce' in text:
 701        return 'bigcommerce'
 702    return 'unknown'

1189def aggregate_and_rank(all_code_lists, product_info=None):
1190    """
1191    Aggregates discount codes from multiple sources, tallies their frequencies, and ranks them.
1192
1193    This function flattens the lists of codes provided by different scraping sources 
1194    and calculates how many times each unique code was found. It then ranks the 
1195    codes using one of two strategies:
1196    1. Contextual Ranking: If product metadata (name or brand) is available, it 
1197       uses `filter_codes_by_product()` to group codes into relevant, neutral, 
1198       and irrelevant tiers.
1199    2. Fallback Scoring: If no product context exists, it sorts the codes descending 
1200       based on a combined weight of their frequency (frequency * 2) and their 
1201       baseline heuristic score from `score_code()`.
1202
1203    Args:
1204        all_code_lists (list of list of str): A list containing lists of discount 
1205            codes extracted from various sources.
1206        product_info (dict, optional): Extracted product metadata used for contextual 
1207            ranking. Defaults to None.
1208
1209    Returns:
1210        tuple: A 2-tuple containing:
1211            - ranked (list of str): The aggregated, deduplicated codes sorted by relevance.
1212            - freq (dict): A dictionary mapping each code (str) to the number of 
1213              sources (int) that found it.
1214    """
1215    freq = {}
1216    for codes in all_code_lists:
1217        for code in codes:
1218            freq[code] = freq.get(code, 0) + 1
1219
1220    all_codes = list(freq.keys())
1221
1222    if product_info and (product_info.get('product_name') or product_info.get('product_brand')):
1223        relevant, neutral, irrelevant = filter_codes_by_product(all_codes, product_info)
1224        ranked = relevant + neutral + irrelevant
1225    else:
1226        ranked = sorted(all_codes, key=lambda c: (freq[c] * 2 + score_code(c)), reverse=True)
1227
1228    return ranked, freq
